# 分析

对于一条信息$x,y$，将两者之间的牛的高度$-1$.

考虑差分，将区间修改转化为单点修改

注意，排除重复信息

```cpp
#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
const int N=10004;
int n,r,h;//I省略
int c[N];
pair<int,int> a[N];
int main(){
	scanf("%d%*d%d%d",&n,&h,&r);//忽略I
	for(int i=1;i<=r;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		if(x>y)x^=y^=x^=y;
		a[i].first=x,a[i].second=y;
	}
	sort(a+1,a+r+1);
	for(int i=1;i<=r;i++){
		if(a[i]==a[i-1])continue;//去重
		c[a[i].first+1]--,c[a[i].second]++;
	}
	for(int i=1;i<=n;i++){
		c[i]+=c[i-1];
		printf("%d\n",h+c[i]);
	}
	return 0;
}
```
