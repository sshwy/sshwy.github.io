## A.Prefix Sum
- 写一个表就发现是杨辉三角
- 于是裸的组合数
- 模数是质数，用费马小定理算逆元即可。
```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N=5003,M=5003;
const ll P=998244353;

ll n,m,a[N][M];

ll ksm(ll a,ll b){// 快速幂
	ll res=1;
	while(b){if(b&1)res=res*a%P;
		a=a*a%P,b>>=1;
	}
	return res;
}
ll c(ll a,ll b){// 组合数
	if(a<b)return 0;
	ll res=1,fac=1,d=a-b;
	for(ll i=a;i>b;i--)res=res*i%P;
	for(ll i=1;i<=d;i++)fac=fac*i%P;
	ll inv=ksm(fac,P-2);
	return res*inv%P;
}

ll lucas(ll a,ll b){// 毫无卵用的卢卡斯
	ll res=1;
	while(a||b){res*=c(a%P,b%P);
		a/=P,b/=P;
	}
	return res;
}

int main(){scanf("%lld%lld",&n,&m);
	printf("%lld",lucas(m+n-1,m));
	return 0;
}
```
