# 分析

完全背包模板

```cpp
#include<cstdio>
#define FOR(a,b,c) for(int a=b;a<=c;a++)
using namespace std;
const int N=201;
bool bp[N];
int p[N],cnt;
int n,f[N];
void prime_work(int k){
	bp[0]=bp[1]=1;
	FOR(i,2,k){
		if(!bp[i])p[++cnt]=i;
		FOR(j,1,cnt){
			if(i*p[j]>k)break;
			bp[i*p[j]]=1;
			if(i%p[j]==0)break;
		}
	}
}
int main(){
	prime_work(200);
	f[0]=1;
	FOR(i,1,cnt)FOR(j,p[i],200)f[j]+=f[j-p[i]];
	while(~scanf("%d",&n))printf("%d\n",f[n]);
	return 0;
}
```
