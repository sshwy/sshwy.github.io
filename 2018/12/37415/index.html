const int N=4205;
int n,P,ans,p[N][N][2];
int main(){scanf("%d%d",&n,&P);
    p[1][1][0]=p[1][1][1]=1;
    for(int i=2;i<=n;i++){
        for(int j=1;j<=i;j++){
            p[i][j][1]=((long long)p[i][j-1][1]+p[i-1][j-1][0])%P;
            p[i][j][0]=((long long)p[i][j-1][0]+p[i-1][i-1][1]-p[i-1][j-1][1])%P;
        }
    }
    ans=(((long long)p[n][n][0]+p[n][n][1])%P+P)%P;
    printf("%d",ans);
