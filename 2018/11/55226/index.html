#include<bits/stdc++.h> 
using namespace std;
typedef long long ll;
ll T;

ll phi(ll a){
    int res=a;
    for(ll i=2;i*i<=a;i++)if(a%i==0){
        res=res/i*(i-1);
        while(a%i==0)a/=i;
    }
    if(a>1)res=res*(a-1)/a;
    return res;
}
ll ksm(ll a,ll m,ll md){
    ll res=1;
    while(m){
        if(m&1)res=res*a%md;
        a=a*a%md,m>>=1;
    }
    return res;
}
ll f(ll a){
    if(a==1)return 0;
    ll pi=phi(a);
    return ksm(2,f(pi)+pi,a);
}

int main(){
    scanf("%lld",&T);
    while(T--){
        ll p;
        scanf("%lld",&p);
        printf("%lld\n",f(p));
    }
    return 0;
}
